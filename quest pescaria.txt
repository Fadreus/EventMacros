}

automacro primeiraPesca {
	InInventory "Meat" > 0
	exclusive 1
	NpcNotNear /Ripple/
	ConfigKeyNot pescaria 1Feito
	call {
	call pescar "izlu2dun" "157 63" "1Feito"
	}
}

macro pescar {
	do move $.param1 $.param2 &rand ( 1,6 )
	do talk &npc ( $.param2 )
	do talk resp 0
	do talk resp /Meat/
	do conf -f pescaria $.param3
}
